162 void efd::DependencyBuilderVisitor::visit(NDQOpGen::Ref ref) {
163  // Single qbit gate.
164  if (ref->getQArgs()->getChildNumber() == 1) return;
165 
166  auto gate = getParentGate(ref);
167  auto deps = mDepBuilder.getDepsSet(gate);
168 
169  // Getting the qargs uint32_t representations.
170  std::vector<uint32_t> uidVector;
171  for (auto& childRef : *ref->getQArgs())
172  uidVector.push_back(mDepBuilder.getUId(childRef.get(), gate));
173 
174  // Getting the gate declaration node.
175  Node::Ref node = mMod.getQGate(ref->getId()->getVal());
176  NDGateDecl::Ref gRef = dynCast<NDGateDecl>(node);
177  assert(gRef != nullptr && "There is no quantum gate with this id.");
178 
179  auto& gDeps = mDepBuilder.mLDeps[gRef];
180  Dependencies thisDeps { {}, ref };
181  // For every qarg uint32_t representation
182  for (auto parallelDeps : gDeps) {
183  for (auto dep : parallelDeps) {
184  // Getting the uid's of the qubit interaction (u, v)
185  uint32_t u = uidVector[dep.mFrom];
186  uint32_t v = uidVector[dep.mTo];
187  thisDeps.mDeps.push_back(Dep { u, v });
188  }
189  }
190 
191  if (!thisDeps.isEmpty())
192  deps->push_back(thisDeps);
193  mDepBuilder.mIDeps[ref] = thisDeps;
194 }
