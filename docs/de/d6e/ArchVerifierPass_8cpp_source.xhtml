40 bool ArchVerifierVisitor::visitNDQOp(NDQOp::Ref qop) {
41  bool success;
42 
43  std::vector<uint32_t> qUIds;
44  for (auto& qarg : *qop->getQArgs()) {
45  auto qargStr = qarg->toString(false);
46 
47  if (mArch->hasSId(qargStr)) {
48  qUIds.push_back(mArch->getUId(qargStr));
49  } else {
50  // If there is some quantum operation that uses an inexistent qubit, we already
51  // may return false!
52  return false;
53  }
54  }
55 
56  if (IsCNOTGateCall(qop)) {
57 
58  assert(qUIds.size() == 2 && "CNOT call must have 2 quantum qubits.");
59  success = checkCNOT({ qUIds[0], qUIds[1] });
60 
61  } else if (IsIntrinsicGateCall(qop)) {
62  auto gen = dynCast<NDQOpGen>(qop);
63 
64  assert(gen != nullptr && "Intrinsic call not generic operation!?");
65  assert(gen->isIntrinsic() && "Intrinsic call not marked as intrinsic!?");
66 
67  switch (gen->getIntrinsicKind()) {
68  case NDQOpGen::K_INTRINSIC_SWAP:
69  assert(qUIds.size() == 2 && "SWAP call must have 2 quantum qubits.");
70  success = checkCNOT({ qUIds[0], qUIds[1] }) || checkCNOT({ qUIds[1], qUIds[0] });
71  break;
72 
73  case NDQOpGen::K_INTRINSIC_LCX:
74  assert(qUIds.size() == 3 && "LCX call must have 3 quantum qubits.");
75  success = (checkCNOT({ qUIds[0], qUIds[1] }) || checkCNOT({ qUIds[1], qUIds[0] })) &&
76  (checkCNOT({ qUIds[1], qUIds[2] }) || checkCNOT({ qUIds[2], qUIds[1] }));
77  break;
78 
79  case NDQOpGen::K_INTRINSIC_REV_CX:
80  assert(qUIds.size() == 2 && "Reversal call must have 2 quantum qubits.");
81  success = checkCNOT({ qUIds[1], qUIds[0] });
82  break;
83  }
84  } else {
85  success = true;
86  }
87 
88  return success;
89 }
