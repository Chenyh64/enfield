90 void efd::XbitToNumberVisitor::visit(NDRegDecl::Ref ref) {
91  std::string id = ref->getId()->getVal();
92  IntVal size = ref->getSize()->getVal();
93 
94  mXbitToNumber.gidRegMap[id] = std::vector<uint32_t>();
95 
96  auto mapref = &mXbitToNumber.gidQMap;
97  if (ref->isCReg()) mapref = &mXbitToNumber.gidCMap;
98  uint32_t basen = mapref->size();
99 
100  // For each register declaration, we associate a
101  // number to each possible xbit.
102  //
103  // For example, 'qreg q[5];' generates 'q[0]', 'q[1]', ...
104  for (int i = 0; i < size.mV; ++i) {
105  std::string key = id + "[" + std::to_string(i) +"]";
106  auto ref = NDIdRef::Create(NDId::Create(id), NDInt::Create(std::to_string(i)));
107  auto info = XbitToNumber::XbitInfo { basen + i, toShared(std::move(ref)) };
108  mapref->insert(std::make_pair(key, info));
109  mXbitToNumber.gidRegMap[id].push_back(basen + i);
110  }
111 }
