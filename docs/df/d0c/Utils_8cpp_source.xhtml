395 void efd::InlineGate(QModule::Ref qmod, NDQOp::Ref qop) {
396  std::string gateId = qop->getId()->getVal();
397 
398  auto gate = qmod->getQGate(gateId);
399  assert (!gate->isOpaque() && "Trying to inline opaque gate.");
400 
401  NDGateDecl::Ref gateDecl = dynCast<NDGateDecl>(gate);
402  assert(gateDecl != nullptr && "No gate with such id found.");
403 
404  VarMap varMap;
405 
406  Node::Ref gateQArgs = gateDecl->getQArgs();
407  Node::Ref qopQArgs = qop->getQArgs();
408  for (uint32_t i = 0, e = gateQArgs->getChildNumber(); i < e; ++i)
409  varMap[gateQArgs->getChild(i)->toString()] = qopQArgs->getChild(i);
410 
411  Node::Ref gateArgs = gateDecl->getArgs();
412  Node::Ref qopArgs = qop->getArgs();
413  for (uint32_t i = 0, e = gateArgs->getChildNumber(); i < e; ++i)
414  varMap[gateArgs->getChild(i)->toString()] = qopArgs->getChild(i);
415 
416  // ------ Replacing
417  QArgsReplaceVisitor visitor(varMap);
418  auto gop = uniqueCastForward<NDGOpList>(gateDecl->getGOpList()->clone());
419 
420  // 'stmt' is the node we are going to replace.
421  Node::Ref stmt = nullptr;
422  auto ifstmt = dynCast<NDIfStmt>(qop->getParent());
423  if (ifstmt != nullptr) stmt = ifstmt;
424  else stmt = qop;
425 
426  // Replace the arguments.
427  std::vector<Node::uRef> inlinedNodes;
428  for (auto& op : *gop) {
429  auto qop = std::move(op);
430  // If its parent is an NDIfStmt, we wrap the the operation into
431  // a clone of the if.
432  if (ifstmt != nullptr) {
433  auto ifclone = uniqueCastForward<NDIfStmt>(ifstmt->clone());
434  ifclone->setQOp(uniqueCastForward<NDQOp>(std::move(qop)));
435  qop.reset(ifclone.release());
436  }
437 
438  // The 'visitor' is applied only in the 'qop'.
439  qop->apply(&visitor);
440  inlinedNodes.push_back(std::move(qop));
441  }
442 
443  qmod->replaceStatement(stmt, std::move(inlinedNodes));
444 }
