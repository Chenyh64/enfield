148 void SemanticVerifierVisitor::visitNDQOp(NDQOp::Ref tgtQOp, NDIfStmt::Ref tgtIfStmt) {
149  // Checking all quantum arguments from the current node.
150  std::vector<uint32_t> tgtOpQubits;
151  std::vector<uint32_t> tgtOpCbits;
152 
153  auto tgtQArgs = tgtQOp->getQArgs();
154  uint32_t tgtQArgsChildrem = tgtQArgs->getChildNumber();
155 
156  for (uint32_t i = 0; i < tgtQArgsChildrem; ++i) {
157  auto qarg = tgtQArgs->getChild(i);
158  tgtOpQubits.push_back(mXtoNTgt.getQUId(qarg->toString(false)));
159  }
160 
161  if (tgtIfStmt != nullptr) {
162  for (auto cbit : mXtoNTgt.getRegUIds(tgtIfStmt->getCondId()->getVal()))
163  tgtOpCbits.push_back(getRealTgtCUId(cbit));
164  }
165 
166  // Checking all quantum arguments from the circuit node.
167  std::vector<uint32_t> srcOpQubits;
168  std::vector<uint32_t> srcOpCbits;
169 
170  auto srcCNode = mIt[getSrcUId(tgtOpQubits[0])];
171 
172  if (!srcCNode->isGateNode()) { mSuccess = false; return; }
173 
174  auto srcNode = srcCNode->node();
175  NDQOp::Ref srcQOp = nullptr;
176 
177  if (tgtIfStmt != nullptr && tgtIfStmt->getKind() == srcNode->getKind()) {
178  auto srcIfStmt = static_cast<NDIfStmt*>(srcNode);
179  srcQOp = srcIfStmt->getQOp();
180 
181  for (auto cbit : mXtoNSrc.getRegUIds(srcIfStmt->getCondId()->getVal()))
182  srcOpCbits.push_back(getTgtUId(getRealSrcCUId(cbit)));
183 
184  } else if (tgtIfStmt == nullptr) {
185  srcQOp = dynCast<NDQOp>(srcNode);
186  }
187 
188  if (srcQOp == nullptr) {
189  // Either the current node \em tgtQOp is a NDIfStmt and the circuit one is not or
190  // the other way around.
191  mSuccess = false;
192  return;
193  }
194 
195  auto srcQArgs = srcQOp->getQArgs();
196  uint32_t srcQArgsChildrem = srcQArgs->getChildNumber();
197 
198  for (uint32_t i = 0; i < srcQArgsChildrem; ++i) {
199  uint32_t qubit = mXtoNSrc.getQUId(srcQArgs->getChild(i)->toString(false));
200  srcOpQubits.push_back(getTgtUId(qubit));
201  }
202 
203  // All qubits involved in the current circuit node must also be involved in the current node.
204  for (uint32_t q : srcOpQubits) {
205  mSuccess = mSuccess &&
206  std::find(tgtOpQubits.begin(), tgtOpQubits.end(), q) != tgtOpQubits.end();
207  }
208 
209  // All qubits and cbits have reached this node (and they are not null).
210  auto firstSrcCNode = mIt[getSrcUId(srcOpQubits[0])];
211  auto firstSrcNode = firstSrcCNode->node();
212  mSuccess = mSuccess && firstSrcCNode->isGateNode() && !mReached[firstSrcNode];
213 
214  // All used qubits have reached the same node (there is no instruction that is dependent
215  // of others that is being executed before its dependencies)
216  for (uint32_t i = 1; i < srcQArgsChildrem; ++i) {
217  mSuccess = mSuccess && mIt[getSrcUId(srcOpQubits[i])]->node() == firstSrcNode;
218  }
219 
220  if (srcOpCbits.size() != tgtOpCbits.size())
221  mSuccess = false;
222 
223  if (!mSuccess) return;
224 
225  // If this operation deals with more than one qubit, we assume it deals with exactly two
226  // qubits, and that it is a CNOT gate.
227  if (srcOpQubits.size() > 1) {
228  if (IsCNOTGateCall(srcQOp)) {
229  // Both CNOTs and REV_CNOTS have the same semantic.
230  // The only difference is in the way they are implemented.
231  SemanticCNOT srcCNOT { srcOpQubits[0], srcOpQubits[1] };
232  SemanticCNOT tgtCNOT { tgtOpQubits[0], tgtOpQubits[1] };
233 
234  if (!IsCNOTGateCall(tgtQOp)) {
235  auto tgtQOpGen = dynCast<NDQOpGen>(tgtQOp);
236 
237  // Checking for intrinsics.
238  if (tgtQOpGen && tgtQOpGen->isIntrinsic() &&
239  tgtQOpGen->getIntrinsicKind() == NDQOpGen::K_INTRINSIC_LCX) {
240  tgtCNOT.u = tgtOpQubits[0];
241  tgtCNOT.v = tgtOpQubits[2];
242  }
243  }
244 
245  // Check if, semanticaly, CNOTs are applied to the same qubits in the same order.
246  mSuccess = mSuccess && srcCNOT == tgtCNOT;
247 
248  } else if (instanceOf<NDQOpBarrier>(srcQOp)) {
249 
250  auto src = dynCast<NDQOpBarrier>(srcQOp);
251  auto tgt = dynCast<NDQOpBarrier>(tgtQOp);
252 
253  mSuccess = src->getQArgs()->getChildNumber() == tgt->getQArgs()->getChildNumber();
254 
255  } else {
256  assert(false && "Node is neither CNOT nor Barrier.");
257  }
258 
259  } else {
260  if (tgtIfStmt != nullptr)
261  mSuccess = mSuccess && firstSrcCNode->node()->getKind() == tgtIfStmt->getKind();
262  else
263  mSuccess = mSuccess && firstSrcCNode->node()->getKind() == tgtQOp->getKind();
264 
265  // Checking all real arguments.
266  auto tgtArgs = tgtQOp->getArgs();
267  auto srcArgs = srcQOp->getArgs();
268  mSuccess = mSuccess && tgtArgs->equals(srcArgs);
269  }
270 
271  if (mSuccess) {
272  std::vector<uint32_t> all;
273  std::vector<uint32_t> toBeAdvanced;
274 
275  all.insert(all.begin(), srcOpQubits.begin(), srcOpQubits.end());
276  all.insert(all.begin(), srcOpCbits.begin(), srcOpCbits.end());
277 
278  for (auto q : all) {
279  toBeAdvanced.push_back(getSrcUId(q));
280  }
281 
282  postprocessing(toBeAdvanced);
283  }
284 }

286 void SemanticVerifierVisitor::visit(NDQOpMeasure::Ref ref) {
287  uint32_t tgtQUId = mXtoNTgt.getQUId(ref->getQBit()->toString(false));
288  uint32_t tgtCUId = getRealTgtCUId(mXtoNTgt.getCUId(ref->getCBit()->toString(false)));
289 
290  auto srcCNode = mIt[getSrcUId(tgtQUId)];
291  auto srcNode = dynCast<NDQOpMeasure>(srcCNode->node());
292 
293  if (srcNode != nullptr) {
294  uint32_t srcQUId = mXtoNSrc.getQUId(srcNode->getQBit()->toString(false));
295  uint32_t srcCUId = getRealSrcCUId(mXtoNSrc.getCUId(srcNode->getCBit()->toString(false)));
296 
297  mSuccess = mSuccess && tgtQUId == getTgtUId(srcQUId);
298  mSuccess = mSuccess && tgtCUId == getTgtUId(srcCUId);
299 
300  mSuccess = mSuccess && srcCNode->isGateNode();
301  mSuccess = mSuccess && !mReached[srcNode];
302  mSuccess = mSuccess && mIt[srcQUId]->node() == mIt[srcCUId]->node();
303  mSuccess = mSuccess && srcNode->getKind() == ref->getKind();
304 
305  if (mSuccess) postprocessing({ srcQUId, srcCUId });
306  }

325 void SemanticVerifierVisitor::visit(NDQOpGen::Ref ref) {

327  auto qargs = ref->getQArgs();
328 
329  uint32_t u = mXtoNTgt.getQUId(qargs->getChild(0)->toString(false));

331 
332  uint32_t a = mAssign[u];

353 bool SemanticVerifierPass::run(QModule* tgt) {
354  PassCache::Run<FlattenPass>(mSrc.get());
355 
356  auto inlinePass = InlineAllPass::Create(mBasis);
357  PassCache::Run(mSrc.get(), inlinePass.get());
358 
359  auto cktpass = PassCache::Get<CircuitGraphBuilderPass>(mSrc.get());
360  auto& ckt = cktpass->getData();
361 
362  auto xtonpassSrc = PassCache::Get<XbitToNumberWrapperPass>(mSrc.get());
363  auto xtonpassTgt = PassCache::Get<XbitToNumberWrapperPass>(tgt);
364 
365  mData = true;
366  auto it = ckt.build_iterator();
367  SemanticVerifierVisitor visitor(xtonpassSrc->getData(),
368  xtonpassTgt->getData(),
369  it,
370  mInitial);
371  for (auto it = tgt->stmt_begin(), end = tgt->stmt_end();
372  it != end && mData; ++it) {
373  (*it)->apply(&visitor);
374  mData = mData && visitor.mSuccess;
375  }
376 
377  for (uint32_t i = 0, e = ckt.size(); i < e; ++i) {
378  if (!it[i]->isOutputNode()) { mData = false; break; }
379  }
380 
381  return false;
382 }
